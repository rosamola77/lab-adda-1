public static boolean esPalindromo2(String p) {

	int i = 0;
	int j = p.length();
	Boolean b = true;
	while(j - i > 1 && b) {
		i = i + 1;
		j = j - 1;
		b = p.charAt(i) == p.charAt(j-1);
	}
	return b;
}
